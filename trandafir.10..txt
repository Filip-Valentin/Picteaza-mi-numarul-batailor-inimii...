<div class="poem-content">
    <div class="title-container">
        <span class="small-number">.10</span>
        <h2 class="poem-title">Trandafir</h2>
    </div>
    <span class="author">de Filip Valentin</span>
    
    <!-- Strofa 1 -->
    <div class="stanza">
        <p>Ai cules un trandafir,</p>
        <p>Nu-i deloc unul perfect</p>
        <p>Însă cu un singur zâmbet</p>
        <p>I-ai ascuns orice defect</p>
    </div>

    <!-- Strofa 2 -->
    <div class="stanza">
        <p>Acest roșu trandafir</p>
        <p>Te iubește fără seamă.</p>
        <p>Dar alegi să îl iubești</p>
        <p>Doar atunci când el te cheamă.</p>
    </div>

    <!-- Strofa 3 -->
    <div class="stanza">
        <p>-Te privesc, tu mă privești</p>
        <p>Te sărut printre petale</p>
        <p>Mă gândesc, tu te gândești</p>
        <p>Să alegi o altă floare</p>
    </div>

    <!-- Strofa 4 -->
    <div class="stanza">
        <p>Doritor fiind din fire</p>
        <p>De momente întregi cu tine</p>
        <p>Nu doar de niște fărâme,</p>
        <p>Doar fărâme de iubire</p>
    </div>

    <!-- Strofa 5 -->
    <div class="stanza">
        <p>A încercat dar fără rost</p>
        <p>Să fie un trandafir alb</p>
        <p>Însă atunci când ea venea</p>
        <p>El pe loc se înroșea.</p>
    </div>

    <!-- Strofa 6 -->
    <div class="stanza">
        <p>El tânjea după iubire</p>
        <p>Și ea, dar nu după a lui</p>
        <p>De asta cădeau mereu</p>
        <p>Petalele trandafirului</p>
    </div>

    <!-- Strofa 7 -->
    <div class="stanza">
        <p>-Atunci când sunt lângă tine</p>
        <p>Mâine e mult prea departe</p>
        <p>Hai sa ne iubim de parcă</p>
        <p>Asta e ultima noapte,</p>
    </div>

    <!-- Strofa 8 -->
    <div class="stanza">
        <p>-Tu iubite trandafir,</p>
        <p>De ce tot te ofilesti?</p>
        <p>-Pentru că mă rupi în două</p>
        <p>Doar atunci când mă privești</p>
    </div>

    <!-- Strofa 9 -->
    <div class="stanza">
        <p>Și cădeau petalele,</p>
        <p>Căte una, două...</p>
        <p>Și plecă și ea</p>
        <p>Și începu să plouă</p>
    </div>

    <!-- Strofa 10 -->
    <div class="stanza">
        <p>Se ofilea ușor.</p>
        <p>Lumina încet s-a stins</p>
        <p>Și căzu ultima petală</p>
        <p>A trandafirului</p>
        <p>De un roșu prea aprins...</p>
    </div>
